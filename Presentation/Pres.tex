\only<7>{
\begin{textblock*}{10cm}(1.5cm,5cm) % {block width} (coords)
\begin{itemize}
\item We select a random site
\item We compute the energy with the neighboors : $\textcolor{blue}{E_{old}}$
\item We try to swap the chosen site
\item We compute the energy with the neighboors : $\textcolor{red}{E_{new}}$
\item We accept the swap with a probability :
\vspace{-0.3cm}
\begin{equation*}
p = e^{-\dfrac{\textcolor{blue}{E_{old}} -\textcolor{red}{E_{new}}}{k_B T}}
\end{equation*}



\end{itemize}
\end{textblock*}
